def simple_fcfs_schedule(data):
    machine_schedules = {m: [] for m in data['Machine_ID'].unique()}
    for _, row in data.iterrows():
        machine_schedules[row['Machine_ID']].append((row['Job_ID'], row['Processing_Time']))
    return machine_schedules

def calculate_makespan(schedule):
    max_time = 0
    for machine, jobs in schedule.items():
        total_time = sum(proc_time for _, proc_time in jobs)
        max_time = max(max_time, total_time)
    return max_time

fcfs_makespan = calculate_makespan(fcfs_schedule)

import random

# Map jobs to machine and processing time
jobs = list(zip(schedule_data['Job_ID'], schedule_data['Machine_ID'], schedule_data['Processing_Time']))
job_to_machine = {job[0]: job[1] for job in jobs}
job_to_proc_time = {job[0]: job[2] for job in jobs}

# Create random chromosome
def create_chromosome(jobs):
    chromosome = [job[0] for job in jobs]
    random.shuffle(chromosome)
    return chromosome

# Schedule jobs by chromosome
def schedule_by_chromosome(chromosome):
    machine_time = {m: 0 for m in schedule_data['Machine_ID'].unique()}
    machine_schedule = {m: [] for m in schedule_data['Machine_ID'].unique()}
    for job_id in chromosome:
        m = job_to_machine[job_id]
        start = machine_time[m]
        proc = job_to_proc_time[job_id]
        machine_schedule[m].append((job_id, start, start + proc))
        machine_time[m] += proc
    return machine_schedule, max(machine_time.values())

# Fitness function (minimize makespan)
def fitness(chromosome):
    _, makespan = schedule_by_chromosome(chromosome)
    return makespan

# Selection: top 50%
def selection(population, fitnesses):
    sorted_pop = sorted(zip(population, fitnesses), key=lambda x: x[1])
    return [chrom for chrom, fit in sorted_pop[:len(sorted_pop)//2]]

# Crossover operator (Order Crossover)
def crossover(parent1, parent2):
    size = len(parent1)
    start, end = sorted(random.sample(range(size), 2))
    child = [None]*size
    child[start:end] = parent1[start:end]
    fill_pos = end
    for gene in parent2:
        if gene not in child:
            if fill_pos >= size:
                fill_pos = 0
            child[fill_pos] = gene
            fill_pos += 1
    return child

# Mutation operator (swap mutation)
def mutate(chromosome, mutation_rate=0.05):
    chrom = chromosome[:]
    for i in range(len(chrom)):
        if random.random() < mutation_rate:
            j = random.randint(0, len(chrom) - 1)
            chrom[i], chrom[j] = chrom[j], chrom[i]
    return chrom

def genetic_algorithm(jobs, pop_size=50, generations=100):
    population = [create_chromosome(jobs) for _ in range(pop_size)]
    best_solution = None
    best_fitness = float('inf')
    fitness_history = []

    for gen in range(generations):
        fitnesses = [fitness(chrom) for chrom in population]
        best_gen_fit = min(fitnesses)
        best_gen_sol = population[fitnesses.index(best_gen_fit)]
        if best_gen_fit < best_fitness:
            best_fitness = best_gen_fit
            best_solution = best_gen_sol
        fitness_history.append(best_fitness)

        selected = selection(population, fitnesses)
        next_population = selected[:]
        while len(next_population) < pop_size:
            parents = random.sample(selected, 2)
            child = crossover(parents[0], parents[1])
            child = mutate(child)
            next_population.append(child)
        population = next_population
        print(f'Generation {gen+1}: Best Makespan = {best_fitness}')

    return best_solution, fitness_history

best_solution, fitness_history = genetic_algorithm(jobs, pop_size=50, generations=30)

best_schedule, best_makespan = schedule_by_chromosome(best_solution)

def plot_gantt(schedule):
    colors = plt.cm.tab20.colors
    color_map = {}
    fig, ax = plt.subplots(figsize=(12, 6))
    yticks = []
    ylabels = []
    for i, (machine, jobs) in enumerate(schedule.items()):
        yticks.append(i)
        ylabels.append(machine)
        for job_id, start, end in jobs:
            if job_id not in color_map:
                color_map[job_id] = colors[len(color_map) % len(colors)]
            ax.barh(i, end-start, left=start, color=color_map[job_id])
            ax.text(start+(end-start)/2, i, job_id, ha='center', va='center', color='white', fontsize=7)
    ax.set_yticks(yticks)
    ax.set_yticklabels(ylabels)
    ax.set_xlabel('Time')
    ax.set_ylabel('Machine')
    ax.set_title('Optimized Manufacturing Schedule Gantt Chart')
    plt.show()

# Background and Motivation
Manufacturing companies like Bayer Crop Science face complex challenges in scheduling multiple production jobs across various machines. Efficient scheduling directly impacts operational costs, resource utilization, and delivery timelines. This project addresses the campaign scheduling problem by leveraging data-driven optimization techniques, with the goal to minimize overall production makespan, a key performance indicator.

# Dataset and Features
The project uses a real-world manufacturing dataset containing 1000 jobs with attributes like job ID, machine ID, processing time, machine availability, and operational details. After cleaning and preprocessing to handle missing data and ensure consistency, essential features such as Job_ID, Machine_ID, Processing_Time, and Machine_Availability were extracted to represent the scheduling problem.

# Methodology
1. **Data Preprocessing:** The dataset was loaded, inspected for missing values, cleaned by removing incomplete records, and prepared for algorithmic processing by converting relevant fields to numeric types.
2. **Baseline Scheduling:** A simple First-Come-First-Served (FCFS) heuristic was implemented to provide a baseline scheduling solution and corresponding makespan metric.
3. **Genetic Algorithm (GA) Optimization:** A metaheuristic GA was designed to optimize job sequences on machines. The solution encoded schedules as permutations of job sequences. The fitness function minimized makespan, and evolutionary operators (selection, crossover, mutation) were iteratively applied to generate better schedules.
4. **Visualization:** Optimized schedules were visualized using Gantt charts, providing intuitive insights into job sequencing and machine utilization.

# Results and Insights
The GA consistently outperformed the FCFS baseline, achieving significant reductions in makespan. This demonstrates the value of metaheuristic optimization in tackling combinatorial scheduling problems where exact solutions are computationally prohibitive. Gantt chart visualizations highlighted efficient allocation and sequencing of jobs, supporting interpretability for operational decision-makers.
